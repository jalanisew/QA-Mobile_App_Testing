-- Cart Validation
SELECT product_id, quantity, total_price FROM cart WHERE user_id = 200;

-- Checkout Validation
SELECT order_id, status, total_amount FROM orders WHERE user_id = 200 ORDER BY order_date DESC;

SELECT title, message, deeplink FROM notifications WHERE user_id = 200;
